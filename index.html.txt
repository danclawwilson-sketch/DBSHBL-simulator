let chartObj = null;

async function startSimulation() {
    const log = document.getElementById('logConsole');
    log.innerHTML = "正在獲取幣安數據...";

    let priceData = [];
    try {
        const response = await fetch('https://api.binance.com/api/v3/klines?symbol=ETHUSDT&interval=1d&limit=365');
        const data = await response.json();
        priceData = data.map(d => ({
            date: new Date(d[0]).toLocaleDateString(),
            open: parseFloat(d[1]), high: parseFloat(d[2]), low: parseFloat(d[3]), close: parseFloat(d[4]),
            avg: (parseFloat(d[2]) + parseFloat(d[3]) + parseFloat(d[4])) / 3
        }));
    } catch (e) { log.innerHTML = "失敗: " + e; return; }

    let reserve = parseFloat(document.getElementById('initReserve').value);
    let poolCount = parseInt(document.getElementById('poolCount').value);
    let poolInitSize = parseFloat(document.getElementById('poolSize').value);
    
    // 初始化水池，加入剩餘天數與基準價追蹤
    let pools = Array(poolCount).fill(0).map(() => ({ 
        balance: poolInitSize, 
        remainingDays: 0, 
        basePrice: 0 
    }));
    let ethPool = { amount: 0, cost: 0, remainingDays: 0 };
    
    const dbDays = parseInt(document.getElementById('dbDays').value);
    const dbKO_pct = parseFloat(document.getElementById('dbKO').value) / 100;
    const dbTarget_pct = parseFloat(document.getElementById('dbTarget').value) / 100;
    const dbAPR = parseFloat(document.getElementById('dbAPR').value) / 100;

    const shDays = parseInt(document.getElementById('shDays').value);
    const shTarget_pct = parseFloat(document.getElementById('shTarget').value) / 100;
    const shAPR = parseFloat(document.getElementById('shAPR').value) / 100;

    let equityHistory = [];
    let dateLabels = [];
    log.innerHTML = "";

    for (let i = 0; i < priceData.length; i++) {
        let today = priceData[i];
        let yesterdayClose = (i === 0) ? today.open : priceData[i-1].close;

        // --- 1. SH 策略處理 ---
        if (ethPool.amount > 0) {
            if (ethPool.remainingDays > 0) {
                ethPool.remainingDays--;
            }
            if (ethPool.remainingDays === 0) {
                let shTargetPrice = ethPool.cost * (1 + shTarget_pct);
                if (today.avg >= shTargetPrice) {
                    let finalUSDT = ethPool.amount * shTargetPrice * (1 + (shAPR / 365 * shDays));
                    reserve += finalUSDT;
                    log.innerHTML += `<div class="log-entry"><b>${today.date}</b> <span class="tag tag-sh">SH 結算賣出</span> 匯入備用金 $${finalUSDT.toFixed(2)}</div>`;
                    ethPool.amount = 0; ethPool.cost = 0;
                } else {
                    ethPool.amount *= (1 + (shAPR / 365 * shDays));
                    ethPool.remainingDays = shDays; // 重新掛單
                }
            }
        }

        // --- 2. DB 策略處理 ---
        pools.forEach((p, idx) => {
            // 如果水池空了或金額不足，補血
            if (p.balance < poolInitSize && p.remainingDays === 0) {
                let diff = poolInitSize - p.balance;
                if (reserve >= diff) { reserve -= diff; p.balance = poolInitSize; }
            }

            if (p.balance > 0) {
                if (p.remainingDays > 0) {
                    p.remainingDays--; // 鎖定中
                } else {
                    // 結算日判斷 (如果是剛開始或是剛結束一個週期)
                    if (p.basePrice > 0) {
                        let koPrice = p.basePrice * (1 + dbKO_pct);
                        let targetPrice = p.basePrice * (1 - dbTarget_pct);

                        if (today.avg >= koPrice) {
                            p.balance *= (1 + (dbAPR / 365 * dbDays));
                        } else if (today.avg <= targetPrice) {
                            let ethBought = p.balance / targetPrice;
                            ethPool.amount += ethBought; ethPool.cost = targetPrice;
                            ethPool.remainingDays = shDays; p.balance = 0;
                            log.innerHTML += `<div class="log-entry"><b>${today.date}</b> <span class="tag tag-db">DB 情境一</span> E${idx+1} 買入 ETH</div>`;
                        } else {
                            let halfUSDT = p.balance / 2;
                            let ethBought = halfUSDT / targetPrice;
                            ethPool.amount += ethBought; ethPool.cost = targetPrice;
                            ethPool.remainingDays = shDays; p.balance = halfUSDT;
                            log.innerHTML += `<div class="log-entry"><b>${today.date}</b> <span class="tag tag-db">DB 情境二</span> E${idx+1} 半倉買入</div>`;
                        }
                        p.basePrice = 0; // 結算完成，清除基準價
                    }
                    
                    // 開始新週期
                    if (p.balance > 0) {
                        p.basePrice = yesterdayClose;
                        p.remainingDays = dbDays;
                    }
                }
            }
        });

        let ethValue = ethPool.amount * today.close;
        let totalEquity = reserve + pools.reduce((sum, p) => sum + p.balance, 0) + ethValue;
        equityHistory.push(totalEquity.toFixed(2));
        dateLabels.push(today.date);
        
        document.getElementById('resEthPrice').innerText = '$' + today.avg.toFixed(2);
        document.getElementById('resEthQty').innerText = ethPool.amount.toFixed(4);
        document.getElementById('resTotal').innerText = '$' + totalEquity.toLocaleString(undefined, {maximumFractionDigits:0});
        document.getElementById('resReserve').innerText = '$' + reserve.toLocaleString(undefined, {maximumFractionDigits:0});
        document.getElementById('resETHPool').innerText = '$' + ethValue.toLocaleString(undefined, {maximumFractionDigits:0});
        let totalInit = parseFloat(document.getElementById('initReserve').value) + (poolCount * poolInitSize);
        document.getElementById('resROI').innerText = ((totalEquity - totalInit) / totalInit * 100).toFixed(2) + '%';
    }
    updateChart(dateLabels, equityHistory);
    log.scrollTop = log.scrollHeight;
}

function updateChart(labels, data) {
    const ctx = document.getElementById('mainChart').getContext('2d');
    if (chartObj) chartObj.destroy();
    chartObj = new Chart(ctx, {
        type: 'line',
        data: { labels: labels, datasets: [{ label: '資產價值', data: data, borderColor: '#f3ba2f', pointRadius: 0, fill: true, backgroundColor: 'rgba(243, 186, 47, 0.1)' }] },
        options: { responsive: true, plugins: { legend: { display: false } }, scales: { x: { display: false } } }
    });
}
